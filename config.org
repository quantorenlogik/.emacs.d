** copy-whole-line (like vim)
#+BEGIN_SRC emacs-lisp
(defun copy-whole-line ()
  (interactive)
  (save-excursion
        (kill-new
         (buffer-substring
          (point-at-bol)
          (point-at-eol)))))
(global-set-key (kbd "C-c w l") 'copy-whole-line)

  (aset buffer-display-table ?\^M []))
